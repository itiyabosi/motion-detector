from Cocoa import NSBitmapImageRep
from PIL import Image

class ScreenCapture:
    def capture_window(self, app_name="Xcode"):
        """ç‰¹å®šã‚¢ãƒ—ãƒªã®ã‚¦ã‚£ãƒ³ãƒ‰ã‚¦ã‚’ã‚­ãƒ£ãƒ—ãƒãƒ£"""

            Quartz.kCGNullWindowID
        )

        # ç›®çš„ã®ã‚¢ãƒ—ãƒªã‚’æ¤œç´¢
        target_window = None
        for window in window_list:
            owner_name = window.get('kCGWindowOwnerName', '')
            if app_name in owner_name:
                target_window = window
                break

        if not target_window:
            raise ValueError(f"{app_name} window not found")

        # ã‚¦ã‚£ãƒ³ãƒ‰ã‚¦ã®ä½ç½®ã¨ã‚µã‚¤ã‚ºã‚’å–å¾—
        bounds = target_window['kCGWindowBounds']

        width, height = bounds['Width'], bounds['Height']

        # ç”»é¢ã‚’ã‚­ãƒ£ãƒ—ãƒãƒ£
        region = Quartz.CGRectMake(x, y, width, height)
        image_ref = Quartz.CGWindowListCreateImage(
            region,
            Quartz.kCGWindowListOptionIncludingWindow,
            target_window['kCGWindowNumber'],
            Quartz.kCGWindowImageBoundsIgnoreFraming
        )

        # PIL Imageã«å¤‰æ›
        width = Quartz.CGImageGetWidth(image_ref)
        height = Quartz.CGImageGetHeight(image_ref)
        bytes_per_row = Quartz.CGImageGetBytesPerRow(image_ref)
        data_provider = Quartz.CGImageGetDataProvider(image_ref)
        pixel_data = Quartz.CGDataProviderCopyData(data_provider)

        image_array = np.frombuffer(pixel_data, dtype=np.uint8)
        image_array = image_array.reshape((height, bytes_per_row // 4, 4))
        image_array = image_array[:, :width, :3]  # BGRA -> RGB

        return Image.fromarray(image_array)

import cv2

from PIL import Image
import numpy as np

class UIElementDetector:
    def __init__(self):
        self.min_button_width = 30
        self.max_button_width = 300
        self.min_button_height = 20
        self.max_button_height = 60

    def detect_elements(self, image):
        """ç”»åƒã‹ã‚‰UIè¦ç´ ã‚’æ¤œå‡º"""
        elements = []

        # 1. OCRã§ãƒ†ã‚­ã‚¹ãƒˆè¦ç´ ã‚’æ¤œå‡º
        text_elements = self._detect_text_elements(image)
        elements.extend(text_elements)

        button_elements = self._detect_buttons(image)

        active_elements = self._detect_active_elements(image)

    def _detect_text_elements(self, image):

        # Tesseract OCRå®Ÿè¡Œ
        ocr_data = pytesseract.image_to_data(
            image,
            output_type=pytesseract.Output.DICT,
            config='--psm 11'  # Sparse text mode
        )

        elements = []
        for i in range(len(ocr_data['text'])):
            text = ocr_data['text'][i].strip()
            if not text or ocr_data['conf'][i] < 60:
                continue

            element = {
                'type': 'text',
                'text': text,
                'bbox': (
                    ocr_data['left'][i],
                    ocr_data['top'][i],
                    ocr_data['width'][i],
                    ocr_data['height'][i]
                ),
                'confidence': ocr_data['conf'][i] / 100.0
            }
            elements.append(element)

        return elements

    def _detect_buttons(self, image):
        """ã‚¨ãƒƒã‚¸æ¤œå‡ºã§ãƒœã‚¿ãƒ³ã‚’æ¤œå‡º"""
        # ã‚°ãƒ¬ãƒ¼ã‚¹ã‚±ãƒ¼ãƒ«å¤‰æ›
        gray = cv2.cvtColor(np.array(image), cv2.COLOR_RGB2GRAY)

        # ã‚¨ãƒƒã‚¸æ¤œå‡º
        edges = cv2.Canny(gray, 50, 150)

        # è¼ªéƒ­æ¤œå‡º
        contours, _ = cv2.findContours(
            edges,
            cv2.RETR_EXTERNAL,
            cv2.CHAIN_APPROX_SIMPLE
        )

        elements = []
        for contour in contours:
            x, y, w, h = cv2.boundingRect(contour)

            # ãƒœã‚¿ãƒ³ã‚‰ã—ã„ã‚µã‚¤ã‚ºã‹ãƒã‚§ãƒƒã‚¯
            if (self.min_button_width < w < self.max_button_width and
                self.min_button_height < h < self.max_button_height):

                # çŸ©å½¢åº¦ã‚’ãƒã‚§ãƒƒã‚¯ï¼ˆãƒœã‚¿ãƒ³ã¯çŸ©å½¢ã«è¿‘ã„ï¼‰
                area = cv2.contourArea(contour)
                rect_area = w * h
                if area / rect_area > 0.7:  # 70%ä»¥ä¸ŠãŒçŸ©å½¢
                    element = {
                        'type': 'button',
                        'bbox': (x, y, w, h),
                        'confidence': 0.7

                    elements.append(element)

    def _detect_active_elements(self, image):
        """è‰²æ¤œå‡ºã§ã‚¢ã‚¯ãƒ†ã‚£ãƒ–ãªè¦ç´ ã‚’æ¤œå‡º"""
        # HSVè‰²ç©ºé–“ã«å¤‰æ›
        hsv = cv2.cvtColor(np.array(image), cv2.COLOR_RGB2HSV)

        # é’è‰²ç¯„å›²ï¼ˆmacOSã®ã‚¢ã‚¯ãƒ†ã‚£ãƒ–ãƒœã‚¿ãƒ³ï¼‰
        lower_blue = np.array([100, 50, 50])
        upper_blue = np.array([130, 255, 255])
        blue_mask = cv2.inRange(hsv, lower_blue, upper_blue)

        # ç·‘è‰²ç¯„å›²ï¼ˆå®Ÿè¡Œãƒœã‚¿ãƒ³ï¼‰
        lower_green = np.array([40, 50, 50])
        upper_green = np.array([80, 255, 255])
        green_mask = cv2.inRange(hsv, lower_green, upper_green)

        # ãƒã‚¹ã‚¯ã‚’çµ±åˆ
        combined_mask = cv2.bitwise_or(blue_mask, green_mask)

        # è¼ªéƒ­æ¤œå‡º
        contours, _ = cv2.findContours(
            combined_mask,
            cv2.RETR_EXTERNAL,
            cv2.CHAIN_APPROX_SIMPLE
        )

        elements = []
        for contour in contours:
            x, y, w, h = cv2.boundingRect(contour)
            if w > 10 and h > 10:  # æœ€å°ã‚µã‚¤ã‚º
                element = {
                    'type': 'active',
                    'bbox': (x, y, w, h),
                    'confidence': 0.6
                }
                elements.append(element)

        return elements

    def _merge_overlapping_elements(self, elements):
        """é‡è¤‡ã™ã‚‹è¦ç´ ã‚’çµ±åˆ"""
        if not elements:
            return []

        # IoUï¼ˆIntersection over Unionï¼‰ã§é‡è¤‡åˆ¤å®š
        def calculate_iou(box1, box2):
            x1, y1, w1, h1 = box1
            x2, y2, w2, h2 = box2

            xi1 = max(x1, x2)
            yi1 = max(y1, y2)
            xi2 = min(x1 + w1, x2 + w2)
            yi2 = min(y1 + h1, y2 + h2)

            inter_area = max(0, xi2 - xi1) * max(0, yi2 - yi1)
            box1_area = w1 * h1
            box2_area = w2 * h2
            union_area = box1_area + box2_area - inter_area

            return inter_area / union_area if union_area > 0 else 0

        merged = []
        used = set()

        for i, elem1 in enumerate(elements):
            if i in used:
                continue

            # ãƒ†ã‚­ã‚¹ãƒˆè¦ç´ ã‚’å„ªå…ˆ
            current = elem1.copy()

            for j, elem2 in enumerate(elements[i+1:], i+1):
                if j in used:
                    continue

                iou = calculate_iou(elem1['bbox'], elem2['bbox'])
                if iou > 0.5:  # 50%ä»¥ä¸Šé‡è¤‡
                    # ãƒ†ã‚­ã‚¹ãƒˆæƒ…å ±ã‚’å„ªå…ˆã—ã¦çµ±åˆ
                    if 'text' in elem2 and 'text' not in current:
                        current['text'] = elem2['text']
                    current['confidence'] = max(
                        current['confidence'],
                        elem2['confidence']
                    )
                    used.add(j)

            merged.append(current)
            used.add(i)

        return merged
